  function babypasa_send_e17( $order_id ) {
      if ( get_post_meta( $order_id, '_e17_sent', true ) ) return;

      $order = wc_get_order( $order_id );
      if ( ! $order ) return;

      $first_name    = $order->get_billing_first_name();
      $order_number  = $order->get_order_number();
      $was_paid      = $order->is_paid();
      $order_total   = wc_price( $order->get_total() );
      $payment_slug  = $order->get_payment_method();

      // Items
      $items = [];
      foreach ( $order->get_items() as $item ) {
          $items[] = [
              'name' => $item->get_name(),
              'qty'  => $item->get_quantity(),
          ];
      }

      // Refund info — only if paid online
      $refund_info = null;
      if ( $was_paid ) {
          $refund_info = [
              'amount'   => $order_total,
              'method'   => babypasa_get_refund_label( $payment_slug ),
              'note'     => babypasa_get_refund_note( $payment_slug ),
              'timeline' => '3–5 business days',
          ];
      }

      $shop_url    = home_url( '/' );
      $support_url = 'mailto:support@babypasa.com';

      babypasa_send_email(
          $order->get_billing_email(),
          'An update on your order #' . $order_number
              . ', ' . $first_name,
          'e17-rto-initiated',
          compact(
              'first_name', 'order_number', 'items',
              'refund_info', 'shop_url', 'support_url'
          )
      );

      update_post_meta( $order_id, '_e17_sent', '1' );
  }

====================================================================
IMPORTANT — REFUND NOT PROCESSED YET
====================================================================
E17 tells the customer a refund is AVAILABLE if they want it.
The actual refund is processed only after:

  1. Parcel arrives at warehouse (E20 fires)
  2. Admin inspects and confirms parcel condition
  3. Admin issues refund in WooCommerce order panel
  4. E21 fires — refund confirmed to customer

Do NOT imply refund is automatic or immediate in E17 copy.
"Processed 3–5 days after parcel is received" is the correct framing.

====================================================================
STATE NOTE — DO NOT ADVANCE FROM RTO
====================================================================
Once state = RTO, no further delivery-related emails should fire.
The webhook handler must check: if state is RTO or higher, skip
OUT_FOR_DELIVERY and DELIVERED processing even if those webhooks
arrive late (Upaya sometimes sends stale events).

  // Guard in webhook handler:
  $rto_states = [ 'RTO', 'FOLLOW_UP' ];
  if ( in_array( $current_state, $rto_states, true ) ) {
      // Only allow RTO_COMPLETE to advance
      if ( ! str_contains( $status, 'RTO Complete' ) ) return;
  }
